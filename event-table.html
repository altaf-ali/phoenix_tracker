library(plyr)
library(yaml)
library(tidyverse)

3.1 Loading Dataset

Load Phoenix events.

source('R/phoenix.R')

config <- yaml.load_file("config.yml")
events <- phoenix_load(config, "2017-01-01")

3.2 Latest Events

Show events reported on 2017-02-06.

events %>%
  filter(Date == max(Date)) %>%
  DT::datatable(selection = "single", options = list(scrollX = TRUE))
